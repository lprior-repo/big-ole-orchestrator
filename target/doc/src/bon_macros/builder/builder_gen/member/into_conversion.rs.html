6impl NamedMember {
7    pub(crate) fn param_into(&self, conditional_params: &[ConditionalParams]) -> Result<bool> {
8        // For optional named members the target of the `Into` conversion is the type
9        // inside of the `Option<T>`, not the `Option<T>` itself because we generate
10        // a setter that accepts `T` itself. It also makes this logic stable regardless
11        // if `Option<T>` is used or the member of type `T` has `#[builder(default)]` on it.
12        let scrutinee = self
13            .as_optional_with_ty(&self.orig_ty)
14            .unwrap_or(&self.orig_ty);
15
16        is_into_enabled(self.origin, &self.params, scrutinee, conditional_params)
17    }
18
19    pub(crate) fn setter_method_core_name(&self) -> &syn::Ident {
20        self.params.name.as_ref().unwrap_or(&self.norm_ident)
21    }
22}
23
24impl PositionalFnArgMember {
25    pub(crate) fn param_into(&self, conditional_params: &[ConditionalParams]) -> Result<bool> {
26        // Positional members are never optional. Users must always specify them, so there
27        // is no need for us to look into the `Option<T>` generic parameter, because the
28        // `Option<T>` itself is the target of the into conversion, not the `T` inside it.
29        let scrutinee = self.orig_ty.as_ref();
30
31        is_into_enabled(self.origin, &self.params, scrutinee, conditional_params)
32    }
33
34    pub(crate) fn fn_input_param(
35        &self,
36        conditional_params: &[ConditionalParams],
37    ) -> Result<TokenStream2> {
38        let has_into = self.param_into(conditional_params)?;
39        let norm_ty = &self.norm_ty;
40        let ident = &self.ident;
41
42        let input = if has_into {
43            quote! { #ident: impl Into<#norm_ty> }
44        } else {
45            quote! { #ident: #norm_ty }
46        };
47
48        Ok(input)
49    }
50
51    pub(crate) fn maybe_into_ident_expr(
52        &self,
53        conditional_params: &[ConditionalParams],
54    ) -> Result<TokenStream2> {
55        let has_into = self.param_into(conditional_params)?;
56        let ident = &self.ident;
57
58        let expr = if has_into {
59            quote! {
60                ::core::convert::Into::into(#ident)
61            }
62        } else {
63            ident.to_token_stream()
64        };
65
66        Ok(expr)
67    }
68}
69
70fn is_into_enabled(
71    origin: MemberOrigin,
72    member_params: &MemberParams,
73    scrutinee: &syn::Type,
74    conditional_params: &[ConditionalParams],
75) -> Result<bool> {
76    let verdict_from_defaults = conditional_params
77        .iter()
78        .map(|params| Ok((params, scrutinee.matches(&params.type_pattern)?)))
79        .collect::<Result<Vec<_>>>()?
80        .into_iter()
81        .filter(|(_, matched)| *matched)
82        .any(|(params, _)| params.into.is_present());
83
84    let verdict_from_override = member_params.into.is_present();
85
86    if verdict_from_defaults && verdict_from_override {
87        bail!(
88            &member_params.into.span(),
89            "this `#[builder(into)]` attribute is redundant, because `into` \
90            is already implied for this member via the `#[builder(on(...))]` \
91            at the top of the {}",
92            origin.parent_construct(),
93        );
94    }
95
96    Ok(verdict_from_override || verdict_from_defaults)
97}
