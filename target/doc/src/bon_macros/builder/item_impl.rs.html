9pub(crate) fn generate(mut orig_impl_block: syn::ItemImpl) -> Result<TokenStream2> {
10    if let Some((_, trait_path, _)) = &orig_impl_block.trait_ {
11        bail!(trait_path, "Impls of traits are not supported yet");
12    }
13
14    let (builder_funcs, other_items): (Vec<_>, Vec<_>) =
15        orig_impl_block.items.into_iter().partition(|item| {
16            let fn_item = match item {
17                syn::ImplItem::Fn(fn_item) => fn_item,
18                _ => return false,
19            };
20
21            fn_item
22                .attrs
23                .iter()
24                .any(|attr| attr.path().is_ident("builder"))
25        });
26
27    if builder_funcs.is_empty() {
28        bail!(
29            &Span::call_site(),
30            "There are no #[builder] functions in the impl block, so there is no \
31            need for a #[bon] attribute on the impl block"
32        );
33    }
34
35    orig_impl_block.items = builder_funcs;
36
37    // We do this back-and-forth with normalizing various syntax and saving original
38    // to provide cleaner code generation that is easier to consume for IDEs and for
39    // rust-analyzer specifically.
40    //
41    // For codegen logic we would like to have everything normalized. For example, we
42    // want to assume `Self` is replaced with the original type and all lifetimes are
43    // named, and `impl Traits` are desugared into type parameters.
44    //
45    // However, in output code we want to preserve existing `Self` references to make
46    // sure rust-analyzer highlights them properly. If we just strip `Self` from output
47    // code, then rust-analyzer won't be able to associate what `Self` token maps to in
48    // the input. It would highlight `Self` as an "unresolved symbol"
49    let mut norm_impl_block = orig_impl_block.clone();
50
51    crate::normalization::NormalizeLifetimes.visit_item_impl_mut(&mut norm_impl_block);
52    crate::normalization::NormalizeImplTraits.visit_item_impl_mut(&mut norm_impl_block);
53
54    let mut norm_selfful_impl_block = norm_impl_block.clone();
55
56    crate::normalization::NormalizeSelfTy {
57        self_ty: &norm_impl_block.self_ty.clone(),
58    }
59    .visit_item_impl_mut(&mut norm_impl_block);
60
61    let impl_ctx = Rc::new(ImplCtx {
62        self_ty: norm_impl_block.self_ty,
63        generics: norm_impl_block.generics,
64        allow_attrs: norm_impl_block
65            .attrs
66            .iter()
67            .filter_map(syn::Attribute::to_allow)
68            .collect(),
69    });
70
71    let outputs = orig_impl_block
72        .items
73        .into_iter()
74        .zip(norm_impl_block.items)
75        .map(|(orig_item, norm_item)| {
76            let norm_func = match norm_item {
77                syn::ImplItem::Fn(norm_func) => norm_func,
78                _ => unreachable!(),
79            };
80            let orig_func = match orig_item {
81                syn::ImplItem::Fn(orig_func) => orig_func,
82                _ => unreachable!(),
83            };
84
85            let norm_func = impl_item_fn_into_fn_item(norm_func)?;
86            let orig_func = impl_item_fn_into_fn_item(orig_func)?;
87
88            let meta = orig_func
89                .attrs
90                .iter()
91                .filter(|attr| attr.path().is_ident("builder"))
92                .map(|attr| {
93                    let meta_list = darling::util::parse_attribute_to_meta_list(attr)?;
94                    NestedMeta::parse_meta_list(meta_list.tokens).map_err(Into::into)
95                })
96                .collect::<Result<Vec<_>>>()?
97                .into_iter()
98                .flatten()
99                .collect::<Vec<_>>();
100
101            let params = FuncInputParams::from_list(&meta)?;
102
103            let ctx = FuncInputCtx {
104                orig_func,
105                norm_func,
106                impl_ctx: Some(impl_ctx.clone()),
107                params,
108            };
109
110            Result::<_>::Ok((ctx.adapted_func()?, ctx.into_builder_gen_ctx()?.output()?))
111        })
112        .collect::<Result<Vec<_>>>()?;
113
114    let new_impl_items = outputs.iter().flat_map(|(adapted_func, output)| {
115        let start_func = &output.start_func;
116        [
117            syn::parse_quote!(#start_func),
118            syn::parse_quote!(#adapted_func),
119        ]
120    });
121
122    norm_selfful_impl_block.items = other_items;
123    norm_selfful_impl_block.items.extend(new_impl_items);
124
125    let other_items = outputs.iter().map(|(_, output)| &output.other_items);
126
127    Ok(quote! {
128        #(#other_items)*
129        #norm_selfful_impl_block
130    })
131}

133fn impl_item_fn_into_fn_item(func: syn::ImplItemFn) -> Result<syn::ItemFn> {
134    let syn::ImplItemFn {
135        attrs,
136        vis,
137        defaultness,
138        sig,
139        block,
140    } = func;
141
142    if let Some(defaultness) = &defaultness {
143        bail!(defaultness, "Default functions are not supported yet");
144    }
145
146    Ok(syn::ItemFn {
147        attrs,
148        vis,
149        sig,
150        block: Box::new(block),
151    })
152}
