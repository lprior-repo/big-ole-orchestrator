5pub(crate) fn generate(params: TokenStream2, item: TokenStream2) -> TokenStream2 {

10pub(crate) fn try_generate(params: TokenStream2, item: TokenStream2) -> Result<TokenStream2> {
11    let item: syn::Item = syn::parse2(item)?;
12    let macro_path = syn::parse_quote!(::bon::bon);
13
14    let ctx = ExpandCfg {
15        macro_path,
16        params,
17        item,
18    };
19
20    let (_params, item) = match ctx.expand_cfg()? {
21        ExpansionOutput::Expanded { params, item } => (params, item),
22        ExpansionOutput::Recurse(output) => return Ok(output),
23    };
24
25    match item {
26        syn::Item::Impl(item_impl) => builder::item_impl::generate(item_impl),
27        _ => bail!(
28            &item,
29            "`#[bon]` attribute is expected to be placed on an `impl` block \
30             but it was placed on other syntax instead"
31        ),
32    }
33}
