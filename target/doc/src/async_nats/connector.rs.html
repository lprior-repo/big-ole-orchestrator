74/// Maintains a list of servers and establishes connections.
75pub(crate) struct Connector {
76    /// A map of servers and number of connect attempts.
77    servers: Vec<(ServerAddr, usize)>,
78    options: ConnectorOptions,
79    pub(crate) connect_stats: Arc<Statistics>,
80    attempts: usize,
81    pub(crate) events_tx: tokio::sync::mpsc::Sender<Event>,
82    pub(crate) state_tx: tokio::sync::watch::Sender<State>,
83    pub(crate) max_payload: Arc<AtomicUsize>,
84}
85
86pub(crate) fn reconnect_delay_callback_default(attempts: usize) -> Duration {
87    if attempts <= 1 {
88        Duration::from_millis(0)
89    } else {
90        let exp: u32 = (attempts - 1).try_into().unwrap_or(u32::MAX);
91        let max = Duration::from_secs(4);
92        cmp::min(Duration::from_millis(2_u64.saturating_pow(exp)), max)
93    }
94}

96impl Connector {
97    pub(crate) fn new<A: ToServerAddrs>(
98        addrs: A,
99        options: ConnectorOptions,
100        events_tx: tokio::sync::mpsc::Sender<Event>,
101        state_tx: tokio::sync::watch::Sender<State>,
102        max_payload: Arc<AtomicUsize>,
103        connect_stats: Arc<Statistics>,
104    ) -> Result<Connector, io::Error> {
105        let servers = addrs.to_server_addrs()?.map(|addr| (addr, 0)).collect();
106
107        Ok(Connector {
108            attempts: 0,
109            servers,
110            options,
111            events_tx,
112            state_tx,
113            max_payload,
114            connect_stats,
115        })
116    }

118    pub(crate) async fn connect(&mut self) -> Result<(ServerInfo, Connection), ConnectError> {
119        loop {
120            match self.try_connect().await {
121                Ok(inner) => {
122                    return Ok(inner);
123                }
124                Err(error) => match error.kind() {
125                    ConnectErrorKind::MaxReconnects => {
126                        return Err(ConnectError::with_source(
127                            crate::ConnectErrorKind::MaxReconnects,
128                            error,
129                        ))
130                    }
131                    other => {
132                        self.events_tx
133                            .try_send(Event::ClientError(ClientError::Other(other.to_string())))
134                            .ok();
135                    }
136                },
137            }
138        }
139    }

141    pub(crate) async fn try_connect(&mut self) -> Result<(ServerInfo, Connection), ConnectError> {
142        tracing::debug!(attempt = %self.attempts, "connecting to server");
143        let mut error = None;
144
145        let mut servers = self.servers.clone();
146        if !self.options.retain_servers_order {
147            servers.shuffle(&mut thread_rng());
148            // sort_by is stable, meaning it will retain the order for equal elements.
149            servers.sort_by(|a, b| a.1.cmp(&b.1));
150        }
151
152        for (server_addr, _) in servers {
153            self.attempts += 1;
154            if let Some(max_reconnects) = self.options.max_reconnects {
155                if self.attempts > max_reconnects {
156                    tracing::error!(
157                        attempts = %self.attempts,
158                        max_reconnects = %max_reconnects,
159                        "max reconnection attempts reached"
160                    );
161                    self.events_tx
162                        .try_send(Event::ClientError(ClientError::MaxReconnects))
163                        .ok();
164                    return Err(ConnectError::new(crate::ConnectErrorKind::MaxReconnects));
165                }
166            }
167
168            let duration = (self.options.reconnect_delay_callback)(self.attempts);
169            tracing::debug!(
170                attempt = %self.attempts,
171                server = ?server_addr,
172                delay_ms = %duration.as_millis(),
173                "attempting connection"
174            );
175
176            sleep(duration).await;
177
178            let socket_addrs = server_addr
179                .socket_addrs()
180                .await
181                .map_err(|err| ConnectError::with_source(crate::ConnectErrorKind::Dns, err))?;
182            for socket_addr in socket_addrs {
183                match self
184                    .try_connect_to(
185                        &socket_addr,
186                        server_addr.tls_required(),
187                        server_addr.clone(),
188                    )
189                    .await
190                {
191                    Ok((server_info, mut connection)) => {
192                        if !self.options.ignore_discovered_servers {
193                            for url in &server_info.connect_urls {
194                                let server_addr = url.parse::<ServerAddr>().map_err(|err| {
195                                    ConnectError::with_source(
196                                        crate::ConnectErrorKind::ServerParse,
197                                        err,
198                                    )
199                                })?;
200                                if !self.servers.iter().any(|(addr, _)| addr == &server_addr) {
201                                    tracing::debug!(
202                                        discovered_url = %url,
203                                        "adding discovered server"
204                                    );
205                                    self.servers.push((server_addr, 0));
206                                }
207                            }
208                        }
209
210                        let tls_required = self.options.tls_required || server_addr.tls_required();
211                        let mut connect_info = ConnectInfo {
212                            tls_required,
213                            name: self.options.name.clone(),
214                            pedantic: false,
215                            verbose: false,
216                            lang: LANG.to_string(),
217                            version: VERSION.to_string(),
218                            protocol: Protocol::Dynamic,
219                            user: self.options.auth.username.to_owned(),
220                            pass: self.options.auth.password.to_owned(),
221                            auth_token: self.options.auth.token.to_owned(),
222                            user_jwt: None,
223                            nkey: None,
224                            signature: None,
225                            echo: !self.options.no_echo,
226                            headers: true,
227                            no_responders: true,
228                        };
229
230                        #[cfg(feature = "nkeys")]
231                        if let Some(nkey) = self.options.auth.nkey.as_ref() {
232                            match nkeys::KeyPair::from_seed(nkey.as_str()) {
233                                Ok(key_pair) => {
234                                    let nonce = server_info.nonce.clone();
235                                    match key_pair.sign(nonce.as_bytes()) {
236                                        Ok(signed) => {
237                                            connect_info.nkey = Some(key_pair.public_key());
238                                            connect_info.signature =
239                                                Some(URL_SAFE_NO_PAD.encode(signed));
240                                        }
241                                        Err(_) => {
242                                            tracing::error!("failed to sign nonce with nkey");
243                                            return Err(ConnectError::new(
244                                                crate::ConnectErrorKind::Authentication,
245                                            ));
246                                        }
247                                    };
248                                }
249                                Err(_) => {
250                                    tracing::error!("failed to create key pair from nkey seed");
251                                    return Err(ConnectError::new(
252                                        crate::ConnectErrorKind::Authentication,
253                                    ));
254                                }
255                            }
256                        }
257
258                        #[cfg(feature = "nkeys")]
259                        if let Some(jwt) = self.options.auth.jwt.as_ref() {
260                            if let Some(sign_fn) = self.options.auth.signature_callback.as_ref() {
261                                match sign_fn.call(server_info.nonce.clone()).await {
262                                    Ok(sig) => {
263                                        connect_info.user_jwt = Some(jwt.clone());
264                                        connect_info.signature = Some(sig);
265                                    }
266                                    Err(_) => {
267                                        tracing::error!("failed to sign nonce with JWT callback");
268                                        return Err(ConnectError::new(
269                                            crate::ConnectErrorKind::Authentication,
270                                        ));
271                                    }
272                                }
273                            }
274                        }
275
276                        if let Some(callback) = self.options.auth_callback.as_ref() {
277                            let auth: crate::Auth = callback
278                                .call(server_info.nonce.as_bytes().to_vec())
279                                .await
280                                .map_err(|err| {
281                                    tracing::error!(error = %err, "auth callback failed");
282                                    ConnectError::with_source(
283                                        crate::ConnectErrorKind::Authentication,
284                                        err,
285                                    )
286                                })?;
287                            connect_info.user = auth.username;
288                            connect_info.pass = auth.password;
289                            connect_info.user_jwt = auth.jwt;
290                            #[cfg(feature = "nkeys")]
291                            {
292                                connect_info.signature = auth
293                                    .signature
294                                    .map(|signature| URL_SAFE_NO_PAD.encode(signature));
295                            }
296                            #[cfg(not(feature = "nkeys"))]
297                            {
298                                if auth.signature.is_some() {
299                                    tracing::error!(
300                                        "signature authentication requires 'nkeys' feature"
301                                    );
302                                    return Err(ConnectError::new(
303                                        crate::ConnectErrorKind::Authentication,
304                                    ));
305                                }
306                                connect_info.signature = None;
307                            }
308                            connect_info.auth_token = auth.token;
309                            connect_info.nkey = auth.nkey;
310                        }
311
312                        connection
313                            .easy_write_and_flush(
314                                [ClientOp::Connect(connect_info), ClientOp::Ping].iter(),
315                            )
316                            .await?;
317
318                        match connection.read_op().await? {
319                            Some(ServerOp::Error(err)) => match err {
320                                ServerError::AuthorizationViolation => {
321                                    tracing::error!(error = %err, "authorization violation");
322                                    return Err(ConnectError::with_source(
323                                        crate::ConnectErrorKind::AuthorizationViolation,
324                                        err,
325                                    ));
326                                }
327                                err => {
328                                    tracing::error!(error = %err, "server error during connection");
329                                    return Err(ConnectError::with_source(
330                                        crate::ConnectErrorKind::Io,
331                                        err,
332                                    ));
333                                }
334                            },
335                            Some(_) => {
336                                tracing::info!(
337                                    server = %server_info.port,
338                                    max_payload = %server_info.max_payload,
339                                    "connected successfully"
340                                );
341                                self.attempts = 0;
342                                self.connect_stats.connects.add(1, Ordering::Relaxed);
343                                self.events_tx.try_send(Event::Connected).ok();
344                                self.state_tx.send(State::Connected).ok();
345                                self.max_payload.store(
346                                    server_info.max_payload,
347                                    std::sync::atomic::Ordering::Relaxed,
348                                );
349                                return Ok((server_info, connection));
350                            }
351                            None => {
352                                tracing::error!("connection closed unexpectedly");
353                                return Err(ConnectError::with_source(
354                                    crate::ConnectErrorKind::Io,
355                                    "broken pipe",
356                                ));
357                            }
358                        }
359                    }
360
361                    Err(inner) => {
362                        tracing::debug!(
363                            server = ?server_addr,
364                            error = %inner,
365                            "connection attempt failed"
366                        );
367                        error.replace(inner)
368                    }
369                };
370            }
371        }
372
373        Err(error.unwrap())
374    }

376    pub(crate) async fn try_connect_to(
377        &self,
378        socket_addr: &SocketAddr,
379        tls_required: bool,
380        server_addr: ServerAddr,
381    ) -> Result<(ServerInfo, Connection), ConnectError> {
382        tracing::debug!(
383            socket_addr = %socket_addr,
384            tls_required = %tls_required,
385            "establishing connection"
386        );
387        let mut connection = match server_addr.scheme() {
388            #[cfg(feature = "websockets")]
389            "ws" => {
390                let ws = tokio::time::timeout(
391                    self.options.connection_timeout,
392                    tokio_websockets::client::Builder::new()
393                        .uri(server_addr.as_url_str())
394                        .map_err(|err| {
395                            ConnectError::with_source(crate::ConnectErrorKind::ServerParse, err)
396                        })?
397                        .connect(),
398                )
399                .await
400                .map_err(|_| ConnectError::new(crate::ConnectErrorKind::TimedOut))?
401                .map_err(|err| ConnectError::with_source(crate::ConnectErrorKind::Io, err))?;
402
403                let con = WebSocketAdapter::new(ws.0);
404                Connection::new(Box::new(con), 0, self.connect_stats.clone())
405            }
406            #[cfg(feature = "websockets")]
407            "wss" => {
408                let tls_config =
409                    Arc::new(tls::config_tls(&self.options).await.map_err(|err| {
410                        ConnectError::with_source(crate::ConnectErrorKind::Tls, err)
411                    })?);
412                let tls_connector = tokio_rustls::TlsConnector::from(tls_config);
413                let ws = tokio::time::timeout(
414                    self.options.connection_timeout,
415                    tokio_websockets::client::Builder::new()
416                        .connector(&tokio_websockets::Connector::Rustls(tls_connector))
417                        .uri(server_addr.as_url_str())
418                        .map_err(|err| {
419                            ConnectError::with_source(crate::ConnectErrorKind::ServerParse, err)
420                        })?
421                        .connect(),
422                )
423                .await
424                .map_err(|_| ConnectError::new(crate::ConnectErrorKind::TimedOut))?
425                .map_err(|err| ConnectError::with_source(crate::ConnectErrorKind::Io, err))?;
426                let con = WebSocketAdapter::new(ws.0);
427                Connection::new(Box::new(con), 0, self.connect_stats.clone())
428            }
429            _ => {
430                let tcp_stream = tokio::time::timeout(
431                    self.options.connection_timeout,
432                    TcpStream::connect(socket_addr),
433                )
434                .await
435                .map_err(|_| ConnectError::new(crate::ConnectErrorKind::TimedOut))??;
436                tcp_stream.set_nodelay(true)?;
437
438                Connection::new(
439                    Box::new(tcp_stream),
440                    self.options.read_buffer_capacity.into(),
441                    self.connect_stats.clone(),
442                )
443            }
444        };
445
446        let tls_connection = |connection: Connection| async {
447            tracing::debug!("upgrading connection to TLS");
448            let tls_config = Arc::new(
449                tls::config_tls(&self.options)
450                    .await
451                    .map_err(|err| ConnectError::with_source(crate::ConnectErrorKind::Tls, err))?,
452            );
453            let tls_connector = tokio_rustls::TlsConnector::from(tls_config);
454
455            let domain = rustls_webpki::types::ServerName::try_from(server_addr.host())
456                .map_err(|err| ConnectError::with_source(crate::ConnectErrorKind::Tls, err))?;
457
458            let tls_stream = tls_connector
459                .connect(domain.to_owned(), connection.stream)
460                .await?;
461
462            Ok::<Connection, ConnectError>(Connection::new(
463                Box::new(tls_stream),
464                0,
465                self.connect_stats.clone(),
466            ))
467        };
468
469        // If `tls_first` was set, establish TLS connection before getting INFO.
470        // There is no point in  checking if tls is required, because
471        // the connection has to be be upgraded to TLS anyway as it's different flow.
472        if self.options.tls_first && !server_addr.is_websocket() {
473            connection = tls_connection(connection).await?;
474        }
475
476        let op = connection.read_op().await?;
477        let info = match op {
478            Some(ServerOp::Info(info)) => {
479                tracing::debug!(
480                    server_id = %info.server_id,
481                    version = %info.version,
482                    "received server info"
483                );
484                info
485            }
486            Some(op) => {
487                tracing::error!(received_op = ?op, "expected INFO, got different operation");
488                return Err(ConnectError::with_source(
489                    crate::ConnectErrorKind::Io,
490                    format!("expected INFO, got {op:?}"),
491                ));
492            }
493            None => {
494                tracing::error!("expected INFO, got nothing");
495                return Err(ConnectError::with_source(
496                    crate::ConnectErrorKind::Io,
497                    "expected INFO, got nothing",
498                ));
499            }
500        };
501
502        // If `tls_first` was not set, establish TLS connection if it is required.
503        if !self.options.tls_first
504            && !server_addr.is_websocket()
505            && (self.options.tls_required || info.tls_required || tls_required)
506        {
507            connection = tls_connection(connection).await?;
508        };
509
510        Ok((*info, connection))
511    }

518    #[test]
519    fn reconnect_delay_callback_duration() {
520        let duration = reconnect_delay_callback_default(0);
521        assert_eq!(duration.as_millis(), 0);
522
523        let duration = reconnect_delay_callback_default(1);
524        assert_eq!(duration.as_millis(), 0);
525
526        let duration = reconnect_delay_callback_default(4);
527        assert_eq!(duration.as_millis(), 8);
528
529        let duration = reconnect_delay_callback_default(12);
530        assert_eq!(duration.as_millis(), 2048);
531
532        let duration = reconnect_delay_callback_default(13);
533        assert_eq!(duration.as_millis(), 4000);
534
535        // The max (4s) was reached and we shouldn't exceed it, regardless of the no of attempts
536        let duration = reconnect_delay_callback_default(50);
537        assert_eq!(duration.as_millis(), 4000);
538    }
