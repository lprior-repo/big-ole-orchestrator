6//! Supervision management logic
7//!
8//! Supervision is a special notion of "ownership" over actors by a parent (supervisor).
9//! Supervisors are responsible for the lifecycle of a child actor such that they get notified
10//! when a child actor starts, stops, or panics (when possible). The supervisor can then decide
11//! how to handle the event. Should it restart the actor, leave it dead, potentially die itself
12//! notifying the supervisor's supervisor? That's up to the implementation of the [super::Actor]
13
14use std::collections::HashMap;
15use std::sync::Mutex;
16
17use super::actor_cell::ActorCell;
18use super::messages::SupervisionEvent;
19use crate::ActorId;
20
21/// A supervision tree
22#[derive(Default, Debug)]
23pub(crate) struct SupervisionTree {
24    children: Mutex<Option<HashMap<ActorId, ActorCell>>>,
25    supervisor: Mutex<Option<ActorCell>>,
26    #[cfg(feature = "monitors")]
27    monitors: Mutex<Option<HashMap<ActorId, ActorCell>>>,
28}
29
30impl SupervisionTree {
31    /// Push a child into the tere
32    pub(crate) fn insert_child(&self, child: ActorCell) {
33        let mut guard = self.children.lock().unwrap();
34        if let Some(map) = &mut *(guard) {
35            map.insert(child.get_id(), child);
36        } else {
37            *guard = Some(HashMap::from_iter([(child.get_id(), child)]));
38        }
39    }
40
41    /// Remove a specific actor from the supervision tree (e.g. actor died)
42    pub(crate) fn remove_child(&self, child: ActorId) {
43        let mut guard = self.children.lock().unwrap();
44        if let Some(map) = &mut *(guard) {
45            map.remove(&child);
46        }
47    }
48
49    /// Push a parent into the tere
50    pub(crate) fn set_supervisor(&self, parent: ActorCell) {
51        *(self.supervisor.lock().unwrap()) = Some(parent);
52    }
53
54    /// Remove a specific actor from the supervision tree (e.g. actor died)
55    pub(crate) fn clear_supervisor(&self) {
56        *(self.supervisor.lock().unwrap()) = None;
57    }
58
59    /// Try and retrieve the set supervisor
60    pub(crate) fn try_get_supervisor(&self) -> Option<ActorCell> {
61        self.supervisor.lock().unwrap().clone()
62    }
63
64    /// Set a monitor of this supervision tree
65    #[cfg(feature = "monitors")]
66    pub(crate) fn set_monitor(&self, who: ActorCell) {
67        let mut guard = self.monitors.lock().unwrap();
68        if let Some(map) = &mut *guard {
69            map.insert(who.get_id(), who);
70        } else {
71            *guard = Some(HashMap::from_iter([(who.get_id(), who)]))
72        }
73    }
74
75    /// Remove a specific monitor from the supervision tree
76    #[cfg(feature = "monitors")]
77    pub(crate) fn remove_monitor(&self, who: ActorId) {
78        let mut guard = self.monitors.lock().unwrap();
79        if let Some(map) = &mut *guard {
80            map.remove(&who);
81            if map.is_empty() {
82                *guard = None;
83            }
84        }
85    }
86
87    /// Terminate all your supervised children and unlink them
88    /// from the supervision tree since the supervisor is shutting down
89    /// and can't deal with superivison events anyways
90    pub(crate) fn terminate_all_children(&self) {
91        let mut guard = self.children.lock().unwrap();
92        let cells = if let Some(map) = &mut *guard {
93            map.values().cloned().collect()
94        } else {
95            vec![]
96        };
97        *guard = None;
98        // drop the guard to not deadlock on double-link
99        drop(guard);
100        for cell in cells {
101            cell.terminate();
102            cell.clear_supervisor();
103        }
104    }
105
106    /// Stop all the linked children, but does NOT unlink them (stop flow will do that)
107    pub(crate) fn stop_all_children(&self, reason: Option<String>) {
108        self.for_each_child(|cell| {
109            cell.stop(reason.clone());
110        });
111    }
112
113    /// Drain all the linked children, but does NOT unlink them
114    pub(crate) fn drain_all_children(&self) {
115        self.for_each_child(|cell| {
116            _ = cell.drain();
117        });
118    }
119
120    /// Stop all the linked children, but does NOT unlink them (stop flow will do that),
121    /// and wait for them to exit (concurrently)
122    pub(crate) async fn stop_all_children_and_wait(
123        &self,
124        reason: Option<String>,
125        timeout: Option<crate::concurrency::Duration>,
126    ) {
127        let cells = self.get_children();
128        let mut js = crate::concurrency::JoinSet::new();
129        for cell in cells {
130            let lreason = reason.clone();
131            let ltimeout = timeout;
132            js.spawn(async move { cell.stop_and_wait(lreason, ltimeout).await });
133        }
134        // drain the tasks
135        while let Some(res) = js.join_next().await {
136            #[cfg(any(
137                feature = "async-std",
138                all(target_arch = "wasm32", target_os = "unknown")
139            ))]
140            if res.is_err() {
141                panic!("JoinSet join error");
142            }
143            #[cfg(not(any(
144                feature = "async-std",
145                all(target_arch = "wasm32", target_os = "unknown")
146            )))]
147            {
148                match res {
149                    Err(err) if err.is_panic() => std::panic::resume_unwind(err.into_panic()),
150                    Err(err) => panic!("{err}"),
151                    _ => {}
152                }
153            }
154        }
155    }

157    /// Drain all the linked children, but does NOT unlink them
158    pub(crate) async fn drain_all_children_and_wait(
159        &self,
160        timeout: Option<crate::concurrency::Duration>,
161    ) {
162        let cells = self.get_children();
163        let mut js = crate::concurrency::JoinSet::new();
164        for cell in cells {
165            let ltimeout = timeout;
166            js.spawn(async move { cell.drain_and_wait(ltimeout).await });
167        }
168        // drain the tasks
169        while let Some(res) = js.join_next().await {
170            #[cfg(any(
171                feature = "async-std",
172                all(target_arch = "wasm32", target_os = "unknown")
173            ))]
174            if res.is_err() {
175                panic!("JoinSet join error");
176            }
177            #[cfg(not(any(
178                feature = "async-std",
179                all(target_arch = "wasm32", target_os = "unknown")
180            )))]
181            {
182                match res {
183                    Err(err) if err.is_panic() => std::panic::resume_unwind(err.into_panic()),
184                    Err(err) => panic!("{err}"),
185                    _ => {}
186                }
187            }
188        }
189    }

191    /// Determine if the specified actor is a parent of this actor
192    pub(crate) fn is_child_of(&self, id: ActorId) -> bool {
193        if let Some(parent) = &*(self.supervisor.lock().unwrap()) {
194            parent.get_id() == id
195        } else {
196            false
197        }
198    }

200    /// Return all linked children
201    pub(crate) fn get_children(&self) -> Vec<ActorCell> {
202        let guard = self.children.lock().unwrap();
203        if let Some(map) = &*guard {
204            map.values().cloned().collect()
205        } else {
206            vec![]
207        }
208    }
209
210    /// Execute a closure for each child without allocating a Vec.
211    /// This is more efficient when you just need to iterate over children.
212    pub(crate) fn for_each_child<F>(&self, mut f: F)
213    where
214        F: FnMut(&ActorCell),
215    {
216        let guard = self.children.lock().unwrap();
217        if let Some(map) = &*guard {
218            for cell in map.values() {
219                f(cell);
220            }
221        }
222    }

224    /// Send a notification to the supervisor.
225    ///
226    /// Optimized to collect all targets under a single lock, then send outside the lock
227    /// to minimize lock contention.
228    pub(crate) fn notify_supervisor(&self, evt: SupervisionEvent) {
229        // Collect all notification targets under a single lock acquisition
230        #[cfg(feature = "monitors")]
231        let monitor_targets: Vec<ActorCell> = {
232            let guard = self.monitors.lock().unwrap();
233            if let Some(monitors) = &*guard {
234                monitors.values().cloned().collect()
235            } else {
236                Vec::new()
237            }
238        };
239
240        let supervisor_target = {
241            let guard = self.supervisor.lock().unwrap();
242            (*guard).clone()
243        };
244
245        // Send to all monitors (best-effort, outside the lock)
246        #[cfg(feature = "monitors")]
247        if !monitor_targets.is_empty() {
248            for monitor in monitor_targets.iter() {
249                // Clone the event for each monitor (without requiring inner data to be Clone)
250                let monitor_evt = evt.clone_no_data();
251                if monitor.send_supervisor_evt(monitor_evt).is_err() {
252                    // Best-effort delivery - if send fails, remove the monitor
253                    let mut guard = self.monitors.lock().unwrap();
254                    if let Some(monitors) = &mut *guard {
255                        monitors.remove(&monitor.get_id());
256                    }
257                }
258            }
259        }
260
261        // Send to supervisor
262        if let Some(parent) = supervisor_target {
263            _ = parent.send_supervisor_evt(evt);
264        }
265    }

267    /// Retrieve the number of supervised children
268    #[cfg(test)]
269    pub(crate) fn get_num_children(&self) -> usize {
270        let guard = self.children.lock().unwrap();
271        if let Some(map) = &*guard {
272            map.len()
273        } else {
274            0
275        }
276    }
277
278    /// Retrieve the number of supervised children
279    #[cfg(test)]
280    pub(crate) fn get_num_parents(&self) -> usize {
281        usize::from(self.supervisor.lock().unwrap().is_some())
282    }
