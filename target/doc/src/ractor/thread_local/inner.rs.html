43impl ActorCell {
44    pub(crate) fn new_thread_local<TActor>(
45        name: Option<ActorName>,
46    ) -> Result<(Self, ActorPortSet), SpawnErr>
47    where
48        TActor: crate::thread_local::ThreadLocalActor,
49    {
50        let (props, rx1, rx2, rx3, rx4) =
51            crate::actor::actor_properties::ActorProperties::new_thread_local::<TActor>(
52                name.clone(),
53            );
54        let cell = Self {
55            inner: Arc::new(props),
56        };
57
58        #[cfg(feature = "cluster")]
59        {
60            // registry to the PID registry
61            crate::registry::pid_registry::register_pid(cell.get_id(), cell.clone())?;
62        }
63
64        if let Some(r_name) = name {
65            crate::registry::register(r_name, cell.clone())?;
66        }
67
68        Ok((
69            cell,
70            ActorPortSet {
71                signal_rx: rx1,
72                stop_rx: rx2,
73                supervisor_rx: rx3,
74                message_rx: rx4,
75            },
76        ))
77    }
78}

80impl ActorProperties {
81    pub(crate) fn new_thread_local<TActor>(
82        name: Option<ActorName>,
83    ) -> (
84        Self,
85        OneshotReceiver<Signal>,
86        OneshotReceiver<StopMessage>,
87        mpsc::MpscUnboundedReceiver<SupervisionEvent>,
88        mpsc::MpscUnboundedReceiver<MuxedMessage>,
89    )
90    where
91        TActor: crate::thread_local::ThreadLocalActor,
92    {
93        let id = crate::actor::actor_id::get_new_local_id();
94        let (tx_signal, rx_signal) = mpsc::oneshot();
95        let (tx_stop, rx_stop) = mpsc::oneshot();
96        let (tx_supervision, rx_supervision) = mpsc::mpsc_unbounded();
97        let (tx_message, rx_message) = mpsc::mpsc_unbounded();
98        (
99            Self {
100                id,
101                name,
102                status: AtomicU8::new(ActorStatus::Unstarted as u8),
103                signal: Mutex::new(Some(tx_signal)),
104                wait_handler: mpsc::Notify::new(),
105                stop: Mutex::new(Some(tx_stop)),
106                supervision: tx_supervision,
107                message: tx_message,
108                tree: Default::default(),
109                type_id: std::any::TypeId::of::<TActor::Msg>(),
110                #[cfg(feature = "cluster")]
111                supports_remoting: TActor::Msg::serializable(),
112            },
113            rx_signal,
114            rx_stop,
115            rx_supervision,
116            rx_message,
117        )
118    }

209    /// Spawn an actor instantly, not waiting on the actor's `pre_start` routine.
210    /// Returns the `ActorRef` and a JoinHandle which resolves to the inner actor handle
211    /// once the actor has finished its startup attempt (or the startup error).
212    #[allow(clippy::type_complexity)]
213    pub(crate) fn spawn_instant(
214        name: Option<ActorName>,
215        startup_args: TActor::Arguments,
216        spawner: ThreadLocalActorSpawner,
217    ) -> Result<
218        (
219            ActorRef<TActor::Msg>,
220            JoinHandle<Result<JoinHandle<()>, SpawnErr>>,
221        ),
222        SpawnErr,
223    > {
224        let (actor, ports) = Self::new(name.clone())?;
225        let actor_ref = actor.actor_ref.clone();
226        let join_op = crate::concurrency::spawn_named(name.as_deref(), async move {
227            let result = actor.start(ports, startup_args, spawner, None).await;
228            let (_, handle) = result?;
229            Ok(handle)
230        });
231        Ok((actor_ref, join_op))
232    }

260    /// Start the actor immediately, optionally linking to a parent actor (supervision tree)
261    ///
262    /// NOTE: This returned [crate::concurrency::JoinHandle] is guaranteed to not panic (unless the runtime is shutting down perhaps).
263    /// An inner join handle is capturing panic results from any part of the inner tasks, so therefore
264    /// we can safely ignore it, or wait on it to block on the actor's progress
265    ///
266    /// * `ports` - The [ActorPortSet] for this actor
267    /// * `supervisor` - The optional [ActorCell] representing the supervisor of this actor
268    ///
269    /// Returns a [Ok((ActorRef, JoinHandle<()>))] upon successful start, denoting the actor reference
270    /// along with the join handle which will complete when the actor terminates. Returns [Err(SpawnErr)] if
271    /// the actor failed to start
272    #[tracing::instrument(name = "Actor", skip(self, ports, startup_args, supervisor), fields(id = self.id.to_string(), name = self.name))]
273    async fn start(
274        self,
275        ports: ActorPortSet,
276        startup_args: TActor::Arguments,
277        spawner: ThreadLocalActorSpawner,
278        supervisor: Option<ActorCell>,
279    ) -> Result<(ActorRef<TActor::Msg>, JoinHandle<()>), SpawnErr> {
280        // cannot start an actor more than once
281        if self.actor_ref.get_status() != ActorStatus::Unstarted {
282            return Err(SpawnErr::ActorAlreadyStarted);
283        }
284
285        let Self {
286            actor_ref,
287            id,
288            name,
289        } = self;
290
291        actor_ref.set_status(ActorStatus::Starting);
292
293        // setup supervision synchronously
294        if let Some(sup) = &supervisor {
295            actor_ref.link(sup.clone());
296        }
297
298        // Generate the ActorRef which will be returned
299        let spawn_name = name.clone();
300        let myself_ret = actor_ref.clone();
301
302        // run the processing loop, backgrounding the work
303        let builder = Box::new(move || {
304            let handler = TActor::default();
305            async move {
306                // Perform the pre-start routine, crashing immediately if we fail to start
307                let result = Self::do_pre_start(actor_ref.clone(), &handler, startup_args).await;
308                let mut state = match result {
309                    Ok(Ok(state)) => state,
310                    Ok(Err(e)) => {
311                        // cleanup supervision on pre_start failure
312                        if let Some(sup) = &supervisor {
313                            actor_ref.unlink(sup.clone());
314                        }
315                        return Err(SpawnErr::StartupFailed(e));
316                    }
317                    Err(e) => {
318                        // cleanup supervision on panic
319                        if let Some(sup) = &supervisor {
320                            actor_ref.unlink(sup.clone());
321                        }
322                        return Err(e);
323                    }
324                };
325
326                // run the processing loop, backgrounding the work
327                let handle = crate::concurrency::spawn_local(async move {
328                    let myself = actor_ref.clone();
329                    let evt = match Self::processing_loop(
330                        ports, &mut state, &handler, actor_ref, id, name,
331                    )
332                    .await
333                    {
334                        // IMPORTANT: Because the State in ThreadLocalActor's is not Send, we cannot
335                        // construct a boxed state since it can't be sent to the supervisor
336                        Ok(exit_reason) => {
337                            SupervisionEvent::ActorTerminated(myself.get_cell(), None, exit_reason)
338                        }
339                        Err(actor_err) => match actor_err {
340                            ActorErr::Cancelled => SupervisionEvent::ActorTerminated(
341                                myself.get_cell(),
342                                None,
343                                Some("killed".to_string()),
344                            ),
345                            ActorErr::Failed(msg) => {
346                                SupervisionEvent::ActorFailed(myself.get_cell(), msg)
347                            }
348                        },
349                    };
350
351                    // terminate children
352                    myself.terminate();
353
354                    // notify supervisors of the actor's death
355                    myself.notify_supervisor_and_monitors(evt);
356
357                    // unlink superisors
358                    if let Some(sup) = supervisor {
359                        myself.unlink(sup);
360                    }
361
362                    // set status to stopped
363                    myself.set_status(ActorStatus::Stopped);
364                });
365                Ok(handle)
366            }
367            .boxed_local()
368        });
369        let handle = spawner
370            .spawn(builder, spawn_name)
371            .await
372            .map_err(|e| SpawnErr::StartupFailed(e.into()))?;
373
374        Ok((myself_ret, handle))
375    }

377    #[tracing::instrument(name = "Actor", skip(ports, state, handler, myself, _id, _name), fields(id = _id.to_string(), name = _name))]
378    async fn processing_loop(
379        mut ports: ActorPortSet,
380        state: &mut TActor::State,
381        handler: &TActor,
382        myself: ActorRef<TActor::Msg>,
383        _id: ActorId,
384        _name: Option<String>,
385    ) -> Result<Option<String>, ActorErr> {
386        // perform the post-start, with supervision enabled
387        Self::do_post_start(myself.clone(), handler, state)
388            .await?
389            .map_err(ActorErr::Failed)?;
390
391        myself.set_status(ActorStatus::Running);
392        myself.notify_supervisor_and_monitors(SupervisionEvent::ActorStarted(myself.get_cell()));
393
394        let myself_clone = myself.clone();
395
396        let future = async move {
397            // the message processing loop. If we get an exit flag, try and capture the exit reason if there
398            // is one
399            loop {
400                let ActorLoopResult {
401                    should_exit,
402                    exit_reason,
403                    was_killed,
404                } = Self::process_message(&myself, state, handler, &mut ports)
405                    .await
406                    .map_err(ActorErr::Failed)?;
407                // processing loop exit
408                if should_exit {
409                    return Ok((state, exit_reason, was_killed));
410                }
411            }
412        };
413
414        // capture any panics in this future and convert to an ActorErr
415        let loop_done = futures::FutureExt::catch_unwind(AssertUnwindSafe(future))
416            .map_err(|err| ActorErr::Failed(get_panic_string(err)))
417            .await;
418
419        // set status to stopping
420        myself_clone.set_status(ActorStatus::Stopping);
421
422        let (exit_state, exit_reason, was_killed) = loop_done??;
423
424        // if we didn't exit in error mode, call `post_stop`
425        if !was_killed {
426            Self::do_post_stop(myself_clone, handler, exit_state)
427                .await?
428                .map_err(ActorErr::Failed)?;
429        }
430
431        Ok(exit_reason)
432    }

434    /// Process a message, returning the "new" state (if changed)
435    /// along with optionally whether we were signaled mid-processing or not
436    ///
437    /// * `myself` - The current [ActorRef]
438    /// * `state` - The current [Actor::State] object
439    /// * `handler` - Pointer to the [Actor] definition
440    /// * `ports` - The mutable [ActorPortSet] which are the message ports for this actor
441    ///
442    /// Returns a tuple of the next [Actor::State] and a flag to denote if the processing
443    /// loop is done
444    async fn process_message(
445        myself: &ActorRef<TActor::Msg>,
446        state: &mut TActor::State,
447        handler: &TActor,
448        ports: &mut ActorPortSet,
449    ) -> Result<ActorLoopResult, ActorProcessingErr> {
450        match ports.listen_in_priority().await {
451            Ok(actor_port_message) => match actor_port_message {
452                actor_cell::ActorPortMessage::Signal(signal) => Ok(ActorLoopResult::signal(
453                    Self::handle_signal(myself.clone(), signal),
454                )),
455                actor_cell::ActorPortMessage::Stop(stop_message) => {
456                    let exit_reason = match stop_message {
457                        StopMessage::Stop => {
458                            tracing::trace!("Actor {:?} stopped with no reason", myself.get_id());
459                            None
460                        }
461                        StopMessage::Reason(reason) => {
462                            tracing::trace!(
463                                "Actor {:?} stopped with reason '{reason}'",
464                                myself.get_id(),
465                            );
466                            Some(reason)
467                        }
468                    };
469                    Ok(ActorLoopResult::stop(exit_reason))
470                }
471                actor_cell::ActorPortMessage::Supervision(supervision) => {
472                    let future = Self::handle_supervision_message(
473                        myself.clone(),
474                        state,
475                        handler,
476                        supervision,
477                    );
478                    match ports.run_with_signal(future).await {
479                        Ok(Ok(())) => Ok(ActorLoopResult::ok()),
480                        Ok(Err(internal_err)) => Err(internal_err),
481                        Err(signal) => Ok(ActorLoopResult::signal(Self::handle_signal(
482                            myself.clone(),
483                            signal,
484                        ))),
485                    }
486                }
487                actor_cell::ActorPortMessage::Message(MuxedMessage::Message(msg)) => {
488                    let future = Self::handle_message(myself.clone(), state, handler, msg);
489                    match ports.run_with_signal(future).await {
490                        Ok(Ok(())) => Ok(ActorLoopResult::ok()),
491                        Ok(Err(internal_err)) => Err(internal_err),
492                        Err(signal) => Ok(ActorLoopResult::signal(Self::handle_signal(
493                            myself.clone(),
494                            signal,
495                        ))),
496                    }
497                }
498                actor_cell::ActorPortMessage::Message(MuxedMessage::Drain) => {
499                    // Drain is a stub marker that the actor should now stop, we've processed
500                    // all the messages and we want the actor to die now
501                    Ok(ActorLoopResult::stop(Some("Drained".to_string())))
502                }
503            },
504            Err(MessagingErr::ChannelClosed) => {
505                // one of the channels is closed, this means
506                // the receiver was dropped and in this case
507                // we should always die. Therefore we flag
508                // to terminate
509                Ok(ActorLoopResult::signal(Self::handle_signal(
510                    myself.clone(),
511                    Signal::Kill,
512                )))
513            }
514            Err(MessagingErr::InvalidActorType) => {
515                // not possible. Treat like a channel closed
516                Ok(ActorLoopResult::signal(Self::handle_signal(
517                    myself.clone(),
518                    Signal::Kill,
519                )))
520            }
521            Err(MessagingErr::SendErr(_)) => {
522                // not possible. Treat like a channel closed
523                Ok(ActorLoopResult::signal(Self::handle_signal(
524                    myself.clone(),
525                    Signal::Kill,
526                )))
527            }
528        }
529    }

531    async fn handle_message(
532        myself: ActorRef<TActor::Msg>,
533        state: &mut TActor::State,
534        handler: &TActor,
535        mut msg: crate::message::BoxedMessage,
536    ) -> Result<(), ActorProcessingErr> {
537        // panic in order to kill the actor
538        #[cfg(feature = "cluster")]
539        {
540            // A `RemoteActor` will handle serialized messages, without decoding them, forwarding them
541            // to the remote system for decoding + handling by the real implementation. Therefore `RemoteActor`s
542            // can be thought of as a "shim" to a real actor on a remote system
543            if !myself.get_id().is_local() {
544                match msg.serialized_msg {
545                    Some(serialized_msg) => {
546                        return handler
547                            .handle_serialized(myself, serialized_msg, state)
548                            .await;
549                    }
550                    None => {
551                        return Err(From::from(
552                            "`RemoteActor` failed to read `SerializedMessage` from `BoxedMessage`",
553                        ));
554                    }
555                }
556            }
557        }
558
559        // The current [tracing::Span] is retrieved, boxed, and included in every
560        // `BoxedMessage` during the conversion of this `TActor::Msg`. It is used
561        // to automatically continue tracing span nesting when sending messages to Actors.
562        let current_span_when_message_was_sent = msg.span.take();
563
564        // An error here will bubble up to terminate the actor
565        let typed_msg = TActor::Msg::from_boxed(msg)?;
566
567        if let Some(span) = current_span_when_message_was_sent {
568            handler
569                .handle(myself, typed_msg, state)
570                .instrument(span)
571                .await
572        } else {
573            handler.handle(myself, typed_msg, state).await
574        }
575    }

577    fn handle_signal(myself: ActorRef<TActor::Msg>, signal: Signal) -> Option<String> {
578        match &signal {
579            Signal::Kill => {
580                myself.terminate();
581            }
582        }
583        Some(signal.to_string())
584    }
