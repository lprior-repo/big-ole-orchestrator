6//! Concurrency primitives based on `tokio`
7
8use std::future::Future;
9
10/// Represents a task JoinHandle
11pub type JoinHandle<T> = tokio::task::JoinHandle<T>;
12
13/// A duration of time
14pub type Duration = tokio::time::Duration;
15
16/// An instant measured on system time
17pub type Instant = tokio::time::Instant;
18
19/// Sleep the task for a duration of time
20pub async fn sleep(dur: Duration) {
21    tokio::time::sleep(dur).await;
22}
23
24/// An asynchronous interval calculation which waits until
25/// a checkpoint time to tick
26pub type Interval = tokio::time::Interval;
27
28/// Build a new interval at the given duration starting at now
29///
30/// Ticks 1 time immediately
31pub fn interval(dur: Duration) -> Interval {
32    tokio::time::interval(dur)
33}
34
35/// A set of futures to join on, in an unordered fashion
36/// (first-completed, first-served)
37pub type JoinSet<T> = tokio::task::JoinSet<T>;
38
39/// Spawn a task on the executor runtime
40pub fn spawn<F>(future: F) -> JoinHandle<F::Output>
41where
42    F: Future + Send + 'static,
43    F::Output: Send + 'static,
44{
45    spawn_named(None, future)
46}
47
48/// Spawn a task on the executor runtime which will not be moved to other threads
49pub fn spawn_local<F>(future: F) -> JoinHandle<F::Output>
50where
51    F: Future + 'static,
52{
53    tokio::task::spawn_local(future)
54}
55
56/// Spawn a (possibly) named task on the executor runtime
57pub fn spawn_named<F>(name: Option<&str>, future: F) -> JoinHandle<F::Output>
58where
59    F: Future + Send + 'static,
60    F::Output: Send + 'static,
61{
62    #[cfg(tokio_unstable)]
63    {
64        let mut builder = tokio::task::Builder::new();
65        if let Some(name) = name {
66            builder = builder.name(name);
67        }
68        builder.spawn(future).expect("Tokio task spawn failed")
69    }
70
71    #[cfg(not(tokio_unstable))]
72    {
73        let _ = name;
74        tokio::task::spawn(future)
75    }
76}
77
78/// Execute the future up to a timeout
79///
80/// * `dur`: The duration of time to allow the future to execute for
81/// * `future`: The future to execute
82///
83/// Returns [Ok(_)] if the future succeeded before the timeout, [Err(super::Timeout)] otherwise
84pub async fn timeout<F, T>(dur: super::Duration, future: F) -> Result<T, super::Timeout>
85where
86    F: Future<Output = T>,
87{
88    tokio::time::timeout(dur, future)
89        .await
90        .map_err(|_| super::Timeout)
91}

93macro_rules! select {
94        ($($tokens:tt)*) => {{
95            tokio::select! {
96                // Biased ensures that we poll the ports in the order they appear, giving
97                // priority to our message reception operations. See:
98                // https://docs.rs/tokio/latest/tokio/macro.select.html#fairness
99                // for more information
100                biased;
101
102                $( $tokens )*
103            }
104        }}
105    }
