46/// Represents options for the specified job
47#[derive(Debug, PartialEq, Clone)]
48pub struct JobOptions {
49    /// Time job was submitted from the client
50    submit_time: SystemTime,
51    /// Time job was processed by the factory
52    factory_time: SystemTime,
53    /// Time job was sent to a worker
54    worker_time: SystemTime,
55    /// Time-to-live for the job
56    ttl: Option<Duration>,
57    /// The parent span we want to propagate to the worker.
58    /// Spans don't propagate over the wire in networks
59    span: Option<Span>,
60}
61
62impl JobOptions {
63    /// Create a new [JobOptions] instance, optionally supplying the ttl for the job
64    ///
65    /// * `ttl` - The Time-to-live specification for this job, which is the maximum amount
66    ///   of time the job can remain in the factory's (or worker's) queue before being expired
67    ///   and discarded
68    ///
69    /// Returns a new [JobOptions] instance.
70    pub fn new(ttl: Option<Duration>) -> Self {
71        let span = {
72            #[cfg(feature = "message_span_propogation")]
73            {
74                Some(Span::current())
75            }
76            #[cfg(not(feature = "message_span_propogation"))]
77            {
78                None
79            }
80        };
81        Self {
82            submit_time: SystemTime::now(),
83            factory_time: SystemTime::now(),
84            worker_time: SystemTime::now(),
85            ttl,
86            span,
87        }
88    }
89
90    /// Retrieve the TTL for this job
91    pub fn ttl(&self) -> Option<Duration> {
92        self.ttl
93    }
94
95    /// Set the TTL for this job
96    pub fn set_ttl(&mut self, ttl: Option<Duration>) {
97        self.ttl = ttl;
98    }
99
100    /// Time the job was submitted to the factory
101    /// (i.e. the time `cast` was called)
102    pub fn submit_time(&self) -> SystemTime {
103        self.submit_time
104    }
105
106    /// Time the job was dispatched to a worker
107    pub fn worker_time(&self) -> SystemTime {
108        self.worker_time
109    }
110
111    /// Time the job was received by the factory and first either dispatched
112    /// or enqueued to the factory's queue
113    pub fn factory_time(&self) -> SystemTime {
114        self.factory_time
115    }
116
117    /// Clone the [Span] and return it which is attached
118    /// to this [JobOptions] instance.
119    pub fn span(&self) -> Option<Span> {
120        self.span.clone()
121    }
122
123    pub(crate) fn take_span(&mut self) -> Option<Span> {
124        self.span.take()
125    }
126}
127
128impl Default for JobOptions {
129    fn default() -> Self {
130        Self::new(None)
131    }
132}

134#[cfg(feature = "cluster")]
135impl BytesConvertable for JobOptions {
136    fn into_bytes(self) -> Vec<u8> {
137        let submit_time = (self
138            .submit_time
139            .duration_since(std::time::UNIX_EPOCH)
140            .expect("Time went backwards")
141            .as_nanos() as u64)
142            .to_be_bytes();
143        let ttl = self
144            .ttl
145            .map(|t| t.as_nanos() as u64)
146            .unwrap_or(0)
147            .to_be_bytes();
148
149        let mut data = vec![0u8; 16];
150        data[0..8].copy_from_slice(&submit_time);
151        data[8..16].copy_from_slice(&ttl);
152        data
153    }
154
155    fn from_bytes(mut data: Vec<u8>) -> Self {
156        if data.len() != 16 {
157            Self {
158                span: None,
159                ..Default::default()
160            }
161        } else {
162            let ttl_bytes = data.split_off(8);
163
164            let submit_time = u64::from_be_bytes(data.try_into().unwrap()); //Unwrap should be safe since we checked length earlier
165            let ttl = u64::from_be_bytes(ttl_bytes.try_into().unwrap());
166
167            Self {
168                submit_time: std::time::UNIX_EPOCH + Duration::from_nanos(submit_time),
169                ttl: if ttl > 0 {
170                    Some(Duration::from_nanos(ttl))
171                } else {
172                    None
173                },
174                span: None,
175                ..Default::default()
176            }
177        }
178    }
179}

181/// Represents a job sent to a factory
182///
183/// Depending on the [super::Factory]'s routing scheme the
184/// [Job]'s `key` is utilized to dispatch the job to specific
185/// workers.
186#[derive(Builder)]
187pub struct Job<TKey, TMsg>
188where
189    TKey: JobKey,
190    TMsg: Message,
191{
192    /// The key of the job
193    pub key: TKey,
194    /// The message of the job
195    pub msg: TMsg,
196    /// The job's options, mainly related to timing
197    /// information of the job
198    ///
199    /// Default = [JobOptions::default()]
200    #[builder(default = JobOptions::default())]
201    pub options: JobOptions,
202    /// If provided, this channel can be used to block pushes
203    /// into the factory until the factory can "accept" the message
204    /// into its internal processing. This can be used to synchronize
205    /// external threadpools to the Tokio processing pool and prevent
206    /// overloading the unbounded channel which fronts all actors.
207    ///
208    /// The reply channel return [None] if the job was accepted, or
209    /// [Some(`Job`)] if it was rejected & loadshed, and then the
210    /// job may be retried by the caller at a later time (if desired).
211    ///
212    /// Default = [None]
213    pub accepted: Option<RpcReplyPort<Option<Self>>>,
214}
215
216impl<TKey, TMsg> Debug for Job<TKey, TMsg>
217where
218    TKey: JobKey,
219    TMsg: Message,
220{
221    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
222        f.debug_struct("Job")
223            .field("options", &self.options)
224            .field("has_accepted", &self.accepted.is_some())
225            .finish()
226    }
227}

229#[cfg(feature = "cluster")]
230impl<TKey, TMsg> Job<TKey, TMsg>
231where
232    TKey: JobKey,
233    TMsg: Message,
234{
235    fn serialize_meta(self) -> (Vec<u8>, TMsg) {
236        // exactly 16 bytes
237        let options_bytes = self.options.into_bytes();
238        // variable length bytes based on user-defined encoding
239        let key_bytes = self.key.into_bytes();
240        // build the metadata
241        let mut meta = vec![0u8; 16 + key_bytes.len()];
242        meta[0..16].copy_from_slice(&options_bytes);
243        meta[16..].copy_from_slice(&key_bytes);
244        (meta, self.msg)
245    }
246
247    fn deserialize_meta(
248        maybe_bytes: Option<Vec<u8>>,
249    ) -> Result<(TKey, JobOptions), BoxedDowncastErr> {
250        if let Some(mut meta_bytes) = maybe_bytes {
251            let key_bytes = meta_bytes.split_off(16);
252            Ok((
253                TKey::from_bytes(key_bytes),
254                JobOptions::from_bytes(meta_bytes),
255            ))
256        } else {
257            Err(BoxedDowncastErr)
258        }
259    }
260}

262#[cfg(feature = "cluster")]
263impl<TKey, TMsg> Message for Job<TKey, TMsg>
264where
265    TKey: JobKey,
266    TMsg: Message,
267{
268    fn serializable() -> bool {
269        // The job is serializable if the inner-message is serializable. The key and options
270        // are always serializable
271        TMsg::serializable()
272    }
273
274    fn serialize(self) -> Result<crate::message::SerializedMessage, BoxedDowncastErr> {
275        let (meta_bytes, msg) = self.serialize_meta();
276        // A serialized message so as-expected
277        let inner_message = msg.serialize()?;
278
279        match inner_message {
280            crate::message::SerializedMessage::CallReply(_, _) => Err(BoxedDowncastErr),
281            crate::message::SerializedMessage::Call {
282                variant,
283                args,
284                reply,
285                ..
286            } => Ok(crate::message::SerializedMessage::Call {
287                variant,
288                args,
289                reply,
290                metadata: Some(meta_bytes),
291            }),
292            crate::message::SerializedMessage::Cast { variant, args, .. } => {
293                Ok(crate::message::SerializedMessage::Cast {
294                    variant,
295                    args,
296                    metadata: Some(meta_bytes),
297                })
298            }
299        }
300    }
301
302    fn deserialize(bytes: crate::message::SerializedMessage) -> Result<Self, BoxedDowncastErr> {
303        match bytes {
304            crate::message::SerializedMessage::CallReply(_, _) => Err(BoxedDowncastErr),
305            crate::message::SerializedMessage::Cast {
306                variant,
307                args,
308                metadata,
309            } => {
310                let (key, options) = Self::deserialize_meta(metadata)?;
311                let msg = TMsg::deserialize(crate::message::SerializedMessage::Cast {
312                    variant,
313                    args,
314                    metadata: None,
315                })?;
316                Ok(Self {
317                    msg,
318                    key,
319                    options,
320                    accepted: None,
321                })
322            }
323            crate::message::SerializedMessage::Call {
324                variant,
325                args,
326                reply,
327                metadata,
328            } => {
329                let (key, options) = Self::deserialize_meta(metadata)?;
330                let msg = TMsg::deserialize(crate::message::SerializedMessage::Call {
331                    variant,
332                    args,
333                    reply,
334                    metadata: None,
335                })?;
336                Ok(Self {
337                    msg,
338                    key,
339                    options,
340                    accepted: None,
341                })
342            }
343        }
344    }
345}

347impl<TKey, TMsg> Job<TKey, TMsg>
348where
349    TKey: JobKey,
350    TMsg: Message,
351{
352    /// Determine if this job's TTL is expired
353    ///
354    /// Expiration only takes effect prior to the job being
355    /// started execution on a worker.
356    pub fn is_expired(&self) -> bool {
357        if let Some(ttl) = self.options.ttl {
358            self.options.submit_time.elapsed().unwrap() > ttl
359        } else {
360            false
361        }
362    }
363
364    /// Set the time the factory received the job
365    pub(crate) fn set_factory_time(&mut self) {
366        self.options.factory_time = SystemTime::now();
367    }
368
369    /// Set the time the worker began processing the job
370    pub(crate) fn set_worker_time(&mut self) {
371        self.options.worker_time = SystemTime::now();
372    }
373
374    /// Accept the job (if needed, telling the submitter that the job
375    /// was accepted and enqueued to the factory)
376    pub(crate) fn accept(&mut self) {
377        if let Some(port) = self.accepted.take() {
378            let _ = port.send(None);
379        }
380    }
381
382    /// Reject the job. Consumes the job and returns it to the caller (if needed).
383    pub(crate) fn reject(mut self) {
384        if let Some(port) = self.accepted.take() {
385            let _ = port.send(Some(self));
386        }
387    }
388}

390/// The retry strategy for a [RetriableMessage].
391#[derive(Debug)]
392pub enum MessageRetryStrategy {
393    /// Retry the message forever, without limit.
394    ///
395    /// IMPORTANT: This requires that some other mode is provided
396    /// to mark messages as eventually being `completed()`, be it
397    /// discarding or it being successfully handled. Otherwise the
398    /// message may spin-lock in your factory, never succeeding, and
399    /// constantly being retried
400    RetryForever,
401    /// Retry up to the provided number of times
402    Count(usize),
403    /// No retries (or used to track if retries have been used internally)
404    NoRetry,
405}
406
407impl MessageRetryStrategy {
408    fn has_retries(&self) -> bool {
409        match self {
410            Self::RetryForever => true,
411            Self::Count(n) if *n > 0 => true,
412            _ => false,
413        }
414    }
415
416    fn decrement(&self) -> Self {
417        match self {
418            Self::Count(n) if *n > 1 => Self::Count(*n - 1),
419            Self::RetryForever => Self::RetryForever,
420            _ => Self::NoRetry,
421        }
422    }
423}

425/// A retriable message is a job message which will automatically be resubmitted to the factory in the
426/// event of a factory worker dropping the message due to failure (panic or unhandled error). This wraps
427/// the inner message in a struct which captures the drop, and if there's still some retries left,
428/// will reschedule the work to the factory with captured state information.
429///
430/// IMPORTANT CAVEATS:
431///
432/// 1. Regular loadshed events will cause this message to be retried if there is still retries left
433///    and the job isn't expired unless you explicitely call `completed()` in the discard handler.
434/// 2. Consumable types are not well supported here without some wrapping in Option types, which is
435///    because the value is handled everywhere as `&mut ref` due to the drop implementation requiring that
436///    it be so. This means that RPCs using [crate::concurrency::oneshot]s likely won't work without
437///    some real painful ergonomics.
438/// 3. Upon successful handling of the job, you need to mark it as `completed()` at the end of your
439///    handling or discarding logic to state that it shouldn't be retried on drop.
440pub struct RetriableMessage<TKey: JobKey, TMessage: Message> {
441    /// The key to the retriable job
442    pub key: TKey,
443    /// The message, which will be retried until it's completed.
444    pub message: Option<TMessage>,
445    /// The retry strategy
446    pub strategy: MessageRetryStrategy,
447    /// A function which will be executed upon the job's retry flow being executed
448    /// (helpful for logging, etc)
449    ///
450    /// SAFETY: We utilize [std::panic::catch_unwind] here to best-effort prevent
451    /// a potential user-level panic from causing a process abort. However if the handler
452    /// logic panics, then the retry hook won't be executed, but the job will still be
453    /// retried.
454    #[allow(clippy::type_complexity)]
455    pub retry_hook: Option<Arc<dyn Fn(&TKey) + 'static + Send + Sync + RefUnwindSafe>>,
456
457    retry_state: Option<(JobOptions, ActorRef<FactoryMessage<TKey, Self>>)>,
458}
459
460impl<TKey, TMsg> Debug for RetriableMessage<TKey, TMsg>
461where
462    TKey: JobKey,
463    TMsg: Message,
464{
465    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
466        f.debug_struct("RetriableMessage")
467            .field("key", &self.key)
468            .field("strategy", &self.strategy)
469            .field("message", &self.message.is_some())
470            .field("retry_hook", &self.retry_hook.is_some())
471            .field("retry_state", &self.retry_state.is_some())
472            .finish()
473    }
474}

479impl<TKey: JobKey, TMessage: Message> Drop for RetriableMessage<TKey, TMessage> {
480    fn drop(&mut self) {
481        tracing::trace!("Drop handler for retriable message executing {self:?}");
482        if !self.strategy.has_retries() || self.message.is_none() {
483            // no more retries left (None or Some(>0) mean there's still retries left)
484            // or the payload has been consumed
485            return;
486        }
487        let Some((options, factory)) = self.retry_state.as_ref() else {
488            // can't do a retry if the factory and options are not available
489            return;
490        };
491        // construct the new retriable message
492        let msg = Self {
493            key: self.key.clone(),
494            message: self.message.take(),
495            strategy: self.strategy.decrement(),
496            retry_state: Some((options.clone(), factory.clone())),
497            retry_hook: self.retry_hook.take(),
498        };
499        let job = Job {
500            accepted: None, // should have been accepted on the first try (if accepted at all)
501            key: self.key.clone(),
502            msg,
503            options: options.clone(),
504        };
505        // Execute the custom retry hook, if provided
506        if let Some(handler) = job.msg.retry_hook.as_ref() {
507            let key = std::panic::AssertUnwindSafe(&job.key);
508            let f = handler.clone();
509            _ = std::panic::catch_unwind(move || (f)(*key));
510        }
511        tracing::trace!(
512            "A retriable job is being resubmitted to the factory. Number of retries left {:?}",
513            self.strategy
514        );
515        // SAFETY: A silent-drop here is OK should the dispatch to the factory fail. This is
516        // because if a worker died, it would be a silent drop anyways so there's no loss
517        // in functionality
518        _ = factory.cast(FactoryMessage::Dispatch(job));
519    }
520}

522impl<TKey, TMsg> ActorRef<FactoryMessage<TKey, RetriableMessage<TKey, TMsg>>>
523where
524    TKey: JobKey,
525    TMsg: Message,
526{
527    /// When you're talking to a factory, which accepts "retriable" jobs, this
528    /// convenience function sets up the retriable state for you and converts
529    /// your job to a retriable equivalent.
530    ///
531    /// * `job`: The traditional [Job] which will be auto-converted into a [Job] of [RetriableMessage]
532    ///   for you however the `accepted` field, if set, will be dropped as [RetriableMessage]s do not
533    ///   support `accepted` replies, except on the first iteration
534    /// * `strategy`: The [MessageRetryStrategy] to use for this retriable message
535    ///
536    /// Returns the result from the underlying `cast` operation to the factory's [ActorRef].
537    #[allow(clippy::type_complexity)]
538    pub fn submit_retriable_job(
539        &self,
540        job: Job<TKey, TMsg>,
541        strategy: MessageRetryStrategy,
542    ) -> Result<(), Box<crate::MessagingErr<FactoryMessage<TKey, RetriableMessage<TKey, TMsg>>>>>
543    {
544        let job = RetriableMessage::from_job(job, strategy, self.clone());
545        Ok(self.cast(FactoryMessage::Dispatch(job))?)
546    }
547}

549impl<TKey: JobKey, TMessage: Message> RetriableMessage<TKey, TMessage> {
550    /// Construct a new retriable message with the provided number of retries. If the
551    /// retries are [None], that means retry forever. [Some(0)] will mean no retries
552    pub fn new(key: TKey, message: TMessage, strategy: MessageRetryStrategy) -> Self {
553        Self {
554            key,
555            message: Some(message),
556            strategy,
557            retry_state: None,
558            retry_hook: None,
559        }
560    }
561
562    /// Attach a handler which will be executed when the job is retried (resubmitted to
563    /// the factory).
564    pub fn set_retry_hook(&mut self, f: impl Fn(&TKey) + 'static + Send + Sync + RefUnwindSafe) {
565        self.retry_hook = Some(Arc::new(f));
566    }
567
568    /// Convert a regular [Job] into a [RetriableMessage] capturing all the necessary state in order
569    /// to perform retries on drop.
570    ///
571    /// * `job`: The [Job] to convert
572    /// * `strategy`: The [MessageRetryStrategy] to use for this retriable message
573    /// * `factory`: The [ActorRef] of the factory which the job will be submitted to.
574    ///
575    /// Returns a [Job] with message payload of [RetriableMessage] which can be retried should it be dropped prior
576    /// to having `job.msg.completed()` called.
577    pub fn from_job(
578        Job {
579            key, msg, options, ..
580        }: Job<TKey, TMessage>,
581        strategy: MessageRetryStrategy,
582        factory: ActorRef<FactoryMessage<TKey, Self>>,
583    ) -> Job<TKey, Self> {
584        let mut retriable = RetriableMessage::new(key.clone(), msg, strategy);
585        retriable.capture_retry_state(&options, factory);
586        Job::<TKey, Self> {
587            accepted: None,
588            key,
589            msg: retriable,
590            options,
591        }
592    }
593
594    /// Capture the necessary state information in order to perform retries automatically
595    ///
596    /// * `options`: The [JobOptions] of the original job
597    /// * `factory`: The [ActorRef] of the factory the job will be submitted to
598    pub fn capture_retry_state(
599        &mut self,
600        options: &JobOptions,
601        factory: ActorRef<FactoryMessage<TKey, Self>>,
602    ) {
603        self.retry_state = Some((options.clone(), factory));
604    }
605
606    /// Mark this message to not be retried upon being dropped, since it
607    /// was handled successfully.
608    ///
609    /// IMPORTANT: This should be called at the end of the handling logic, prior to returning
610    /// from the worker's message handler AND/OR from the discard logic. If you don't provide a custom
611    /// discard handler for jobs, then they will be auto-retried until a potential TTL is hit (or forever)
612    /// and you may risk a spin-lock where an TTL expired job is submitted back to the factory, expire shedded
613    /// again, dropped, and therefore retried forever. This is especially dangerous if you state that jobs
614    /// should be retried forever.
615    pub fn completed(&mut self) {
616        self.strategy = MessageRetryStrategy::NoRetry;
617        self.message = None;
618    }
619}

690                crate::message::SerializedMessage::Call { variant, args, .. } => {
691                    match variant.as_str() {
692                        "B" => {
693                            let (tx, _rx) = crate::concurrency::oneshot();
694                            Ok(Self::B(
695                                <String as BytesConvertable>::from_bytes(args),
696                                tx.into(),
697                            ))
698                        }
699                        _ => Err(crate::message::BoxedDowncastErr),
700                    }
701                }
702                _ => Err(crate::message::BoxedDowncastErr),
703            }
704        }
705    }
