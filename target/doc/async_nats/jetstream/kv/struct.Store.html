Iterating over each each key individually

use futures_util::{StreamExt, TryStreamExt};
let client = async_nats::connect("demo.nats.io:4222").await?;

    .create_key_value(async_nats::jetstream::kv::Config {

        history: 10,
        ..Default::default()

    .await?;
